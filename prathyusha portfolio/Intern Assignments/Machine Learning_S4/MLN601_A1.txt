# Outlier Detection

# Identify outliers using the IQR method
Q1 = wine_data['quality'].quantile(0.25)
Q3 = wine_data['quality'].quantile(0.75)

lower_bound = Q1 - 3 * IQR
upper_bound = Q3 + 3 * IQR

wine_data = wine_data[(wine_data['quality'] >= lower_bound) & (wine_data['quality'] <= upper_bound)]

# Count Values
def count_values_table(wine_data):
    count_val = wine_data.value_counts()
    count_val_percent = 100 * wine_data.value_counts() / len(wine_data)
    count_val_table = pd.concat([count_val, count_val_percent.round(1)], axis=1)
    count_val_table_ren_columns = count_val_table.rename(columns={0: 'Count Values', 1: '% of Total Values'})
    return count_val_table_ren_columns

# Histogram
def hist_chart(wine_data, col):

    plt.hist(wine_data[col].dropna(), edgecolor='k')
    plt.xlabel(col)
    plt.ylabel('Number of Entries')
    plt.title('Distribution of ' + col)
